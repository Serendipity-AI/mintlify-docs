
export const MarkdownToPDFPlayground = () => {
  const [markdown, setMarkdown] = React.useState('');
  const [invoice, setInvoice] = React.useState(null);
  const [status, setStatus] = React.useState('');
  const [showModal, setShowModal] = React.useState(false);

  const stubbedInvoiceAPI = async () => ({
    invoice: 'lnbc1-sample-invoice-1234567890abcdefg',
    paymentHash: 'abc123',
  });

  const stubbedPaymentPollingAPI = async () => {
    await new Promise(r => setTimeout(r, 5000));
    return { paid: true, pdfUrl: '/stubbed.pdf' };
  };

  const convert = async () => {
    const { invoice, paymentHash } = await stubbedInvoiceAPI();
    setInvoice(invoice);
    setShowModal(true);
    setStatus('Waiting for payment...');

    setTimeout(async () => {
      const result = await stubbedPaymentPollingAPI(paymentHash);
      if (result.paid) {
        setStatus('Payment received! Downloading PDF...');
        setShowModal(false);
        const link = document.createElement('a');
        link.href = result.pdfUrl;
        link.download = 'converted.pdf';
        link.click();
        setInvoice(null);
        setStatus('');
      }
    }, 3000);
  };

  return (
    <div style={{
      border: 'none',
      borderRadius: '8px',
      padding: '0px',
      marginTop: '1rem',
      position: 'relative'
    }}>
      <h2 style={{ fontSize: '1.5rem', marginBottom: '1rem' }}>Playground</h2>
      <textarea
        style={{
          width: '100%',
          height: '120px',
          fontFamily: 'sans-serif',
          fontSize: '1rem',
          marginBottom: '1rem',
          padding: '12px',
          boxSizing: 'border-box'
        }}
        value={markdown}
        onChange={(e) => setMarkdown(e.target.value)}
        placeholder="Copy and paste your markdown to convert here."
      />
      <br />
      <button
        onClick={convert}
        disabled={!!invoice}
        style={{
          backgroundColor: '#4D42F3',
          color: 'white',
          padding: '10px 16px',
          border: 'none',
          borderRadius: '4px',
          cursor: 'pointer'
        }}
      >
        Convert to PDF
      </button>

      {showModal && invoice && (
        <div style={{
          position: 'fixed',
          top: 0,
          left: 0,
          width: '100vw',
          minHeight: '100vh',            
          backgroundColor: 'rgba(0, 0, 0, 0.6)',
          display: 'flex',
          alignItems: 'center',             
          justifyContent: 'center',          
          boxSizing: 'border-box',          
          padding: '8px',    
          zIndex: 1000             
        }}>
          <div style={{
            backgroundColor: 'white',
            padding: '24px',
            borderRadius: '8px',
            textAlign: 'center',
            maxWidth: '360px',
            boxShadow: '0 8px 20px rgba(0,0,0,0.2)'
          }}>
            <h3 style={{ marginBottom: '0.5rem', color: 'black' }}>Scan to Pay</h3>
            <p style={{ fontSize: '0.9rem', color: '#333', marginBottom: '1rem' }}>
              This is your Lightning invoice. Use any wallet that supports Lightning to pay.
            </p>
            <div style={{ display: 'flex', justifyContent: 'center', marginBottom: '1rem' }}>
              <img
                src={`https://api.qrserver.com/v1/create-qr-code/?size=180x180&data=${encodeURIComponent(invoice)}`}
                alt="Lightning QR code"
              />
            </div>
            <div style={{
              display: 'flex',
              alignItems: 'center',
              backgroundColor: '#f5f5f5',
              padding: '10px',
              borderRadius: '4px',
              marginBottom: '1rem',
              fontFamily: 'monospace',
              fontSize: '0.85rem',
              wordBreak: 'break-all'
            }}>
              <div style={{ flexGrow: 1, paddingRight: '8px' }}>
                {invoice}
              </div>
              <button
                onClick={() => {
                  navigator.clipboard.writeText(invoice);
                  setStatus('Copied!');
                  setTimeout(() => setStatus('Waiting for payment...'), 1000);
                }}
                style={{
                  backgroundColor: 'transparent',
                  border: 'none',
                  padding: 0,
                  cursor: 'pointer',
                  display: 'flex',
                  alignItems: 'center',
                  justifyContent: 'center'
                }}
                aria-label="Copy invoice"
              >
                <svg
                  xmlns="http://www.w3.org/2000/svg"
                  width="16"
                  height="16"
                  fill="none"
                  viewBox="0 0 24 24"
                  stroke="currentColor"
                  strokeWidth="2"
                >
                  <rect x="9" y="9" width="13" height="13" rx="2" ry="2" />
                  <path d="M5 15H4a2 2 0 0 1-2-2V4a2 2 0 0 1 2-2h9a2 2 0 0 1 2 2v1" />
                </svg>
              </button>
            </div>
            <div style={{ color: '#555', marginBottom: '0.5rem' }}>{status}</div>
            <button
              onClick={() => {
                setShowModal(false);
                setInvoice(null);
                setStatus('');
                setIsConverting(false);
              }}
              style={{
                backgroundColor: '#eee',
                padding: '8px 14px',
                border: 'none',
                borderRadius: '4px',
                cursor: 'pointer'
              }}
            >
              Cancel
            </button>
          </div>
        </div>
      )}

      {status && !showModal && (
        <p style={{ marginTop: '1rem', color: '#666' }}>{status}</p>
      )}
    </div>
  );
};

<MarkdownToPDFPlayground />
