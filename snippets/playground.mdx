export const MarkdownToPDFPlayground = () => {
  const [markdown, setMarkdown] = useState('');
  const [invoice, setInvoice] = useState(null);
  const [qrSvg, setQrSvg] = useState('');
  const [status, setStatus] = useState('');
  const [showModal, setShowModal] = useState(false);
  const [paymentPollPayload, setPaymentPollPayload] = useState(null);
  const [isLoading, setIsLoading] = useState(false);
  const isCancelledRef = useRef(false);

  const apiServer = 'https://intelligence-api-qa.ent.sdy.ai';
  const apiPath = '/v1/document/l402/markdown';

  const convert = () => {
    isCancelledRef.current = false;
    setIsLoading(true);

    const payload = {
      data: {
        text_body: markdown,
        meta: {
          title: 'Playground demonstration',
          date: new Date().toISOString()
        }
      },
      options: {
        document_name: 'demonstration.pdf'
      }
    };

    setInvoice(null);
    setQrSvg('');
    setStatus('Requesting conversion...');
    setPaymentPollPayload(payload);

    const doFetch = async () => {
      try {
        const response = await fetch(apiServer + apiPath, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify(payload)
        });

        if (isCancelledRef.current) return;

        if (response.status === 402) {
          const json = await response.json();
          const { payment_request, payment_qr_svg } = json;
          setInvoice(payment_request);
          setQrSvg(payment_qr_svg);
          setShowModal(true);
          setStatus('Waiting for payment...');
          pollPayment(payload);
        } else if (response.status === 200) {
          const data = await response.json();
          const { path } = data;
          setStatus("Payment received! Generating your PDF...");
          pollDocument(apiServer + path);
        } else {
          setStatus('Unexpected response. Please try again.');
          setIsLoading(false);
        }
      } catch (err) {
        console.error(err);
        setStatus('Something went wrong.');
        setIsLoading(false);
      }
    };

    doFetch();
  };

  const pollPayment = (payload) => {
    const poll = async () => {
      try {
        if (isCancelledRef.current) return;

        const res = await fetch(apiServer + apiPath, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify(payload)
        });

        if (isCancelledRef.current) return;

        if (res.status === 402) {
          setTimeout(poll, 3000);
        } else if (res.status === 200 || res.status === 201) {
          const { path } = await res.json();
          setStatus("Payment received! Generating your PDF...");
          pollDocument(apiServer + path);
        } else {
          setStatus('Unexpected response during payment polling.');
          setIsLoading(false);
        }
      } catch (err) {
        console.error(err);
        setStatus('Polling error. Please refresh and try again.');
        setIsLoading(false);
      }
    };

    poll();
  };

  const pollDocument = (documentStatusUrl) => {
    const poll = async () => {
      try {
        if (isCancelledRef.current) return;

        const res = await fetch(documentStatusUrl);

        if (res.status !== 200) {
          setStatus("PDF generation failed.");
          setIsLoading(false);
          return;
        }

        const { status: docStatus, path } = await res.json();

        if (isCancelledRef.current) return;

        if (docStatus !== "Done") {
          setStatus("Waiting for your PDF to generate...");
          setTimeout(poll, 3000);
          return;
        }

        const finalRes = await fetch(apiServer + path);

        if (!finalRes.ok) {
          setStatus("Failed to retrieve final PDF link.");
          setIsLoading(false);
          return;
        }

        const { url } = await finalRes.json();
        const fileRes = await fetch(url);
        const blob = await fileRes.blob();
        const objectUrl = URL.createObjectURL(blob);

        const link = document.createElement("a");
        link.href = objectUrl;
        link.download = "demonstration.pdf";
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        URL.revokeObjectURL(objectUrl);

        setStatus("Download complete");
        setIsLoading(false);
      } catch (err) {
        console.error(err);
        setStatus("Document polling error.");
        setIsLoading(false);
      }
    };

    poll();
  };

  const copyInvoice = () => {
    navigator.clipboard.writeText(invoice);
    setStatus('Copied!');
    setTimeout(() => setStatus('Waiting for payment...'), 1000);
  };

  const cancelPayment = () => {
    isCancelledRef.current = true;
    setShowModal(false);
    setInvoice(null);
    setQrSvg('');
    setStatus('');
    setIsLoading(false);
  };

  return (
    <div>
      <h2 className="text-2xl font-bold mb-2 text-zinc-900 dark:text-zinc-100">Playground</h2>
      <div className="border border-zinc-200 dark:border-zinc-800 rounded-2xl p-6 mt-6">
        <p className="text-sm text-zinc-600 dark:text-zinc-500 mt-0">
          Try it out here. Copy and paste your markdown and then press the button to receive a Lightning payment invoice, allowing you to generate a PDF using any Bitcoin wallet supporting Lightning payments.
        </p>
        <textarea 
          className="w-full h-32 font-mono text-base mb-4 p-3 rounded-2xl border border-zinc-200 dark:border-zinc-700 bg-zinc-50 dark:bg-zinc-900 text-zinc-900 dark:text-zinc-100 placeholder-zinc-400 dark:placeholder-zinc-500 focus:outline-none focus:border-indigo-600 focus:ring-2 focus:ring-indigo-200 dark:focus:border-indigo-400 dark:focus:ring-indigo-900 transition text-sm"
          value={markdown}
          onChange={(e) => setMarkdown(e.target.value)}
          placeholder="Copy and paste your markdown to convert here."
        />
        <button
          onClick={convert}
          disabled={!markdown || isLoading}
          className="px-6 py-3 rounded-2xl font-semibold transition w-full
            bg-indigo-600 dark:bg-indigo-700 text-white shadow-sm hover:bg-indigo-800 dark:hover:bg-indigo-400
            disabled:bg-zinc-200 dark:disabled:bg-zinc-800 disabled:text-zinc-500 dark:disabled:text-zinc-400
            disabled:opacity-50 disabled:shadow-none disabled:cursor-not-allowed flex items-center justify-center gap-2"
        >
          {isLoading ? (
            <>
              <div className="w-5 h-5 border-2 border-zinc-500 border-t-transparent rounded-full animate-spin" />
              Processing...
            </>
          ) : (
            'Convert to PDF'
          )}
        </button>
      </div>

      {showModal && invoice && qrSvg && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/60 dark:bg-black/40 backdrop-blur-sm px-2">
          <div
            className="bg-white dark:bg-zinc-900 p-5 rounded-2xl shadow-2xl w-[95vw] max-w-full sm:max-w-md text-center border border-zinc-200 dark:border-zinc-800 flex flex-col space-y-3 animate-[swipefadeup_0.35s_ease]"
            style={{ minWidth: 0 }}
          >
            <h3 className="text-2xl font-bold text-zinc-900 dark:text-zinc-100 mb-0 mt-1">Scan to Pay</h3>
            <p className="text-sm text-zinc-600 dark:text-zinc-300 mb-0">
              This is your Lightning invoice.<br/>Use any wallet that supports Lightning to pay.
            </p>
            <div className="flex justify-center">
              <img
                src={qrSvg}
                alt="Lightning QR code"
                className="w-44 h-44 bg-white rounded border border-zinc-100 dark:border-zinc-800"
              />
            </div>
            <div className="flex items-center bg-zinc-50 dark:bg-zinc-800 border border-zinc-200 dark:border-zinc-700 rounded-md px-3 py-2 font-mono text-[8px] text-zinc-800 dark:text-zinc-200">
              <div className="flex-1 break-all pr-2 leading-tight">{invoice}</div>
              <button
                onClick={copyInvoice}
                className="p-1 rounded hover:bg-indigo-100 dark:hover:bg-indigo-900 transition"
                aria-label="Copy invoice"
              >
                <svg
                  xmlns="http://www.w3.org/2000/svg"
                  width="16"
                  height="16"
                  fill="none"
                  viewBox="0 0 24 24"
                  stroke="currentColor"
                  strokeWidth="2"
                  className="text-indigo-700 dark:text-indigo-600"
                >
                  <rect x="9" y="9" width="13" height="13" rx="2" ry="2" />
                  <path d="M5 15H4a2 2 0 0 1-2-2V4a2 2 0 0 1 2-2h9a2 2 0 0 1 2 2v1" />
                </svg>
              </button>
            </div>
            <div className="text-indigo-700 dark:text-indigo-600 min-h-[1.5em]">{status}</div>
            <button
              onClick={cancelPayment}
              className="w-full px-4 py-2 rounded-2xl bg-zinc-100 dark:bg-zinc-800 text-zinc-700 dark:text-zinc-200 font-medium hover:bg-zinc-200 dark:hover:bg-zinc-700 transition"
            >
              {status === 'Download Complete!' ? 'OK' : 'Cancel'}
            </button>
          </div>
          <style>
            {`
              @keyframes swipefadeup {
                from { opacity: 0; transform: translateY(32px);}
                to { opacity: 1; transform: translateY(0);}
              }
              @keyframes swipefadedown {
                from { opacity: 1; transform: translateY(0);}
                to { opacity: 0; transform: translateY(32px);}
              }
            `}
          </style>
        </div>
      )}
    </div>
  );
};
