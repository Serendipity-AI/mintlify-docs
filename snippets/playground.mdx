export const MarkdownToPDFPlayground = () => {
  const [markdown, setMarkdown] = useState('');
  const [invoice, setInvoice] = useState(null);
  const [status, setStatus] = useState('');
  const [showModal, setShowModal] = useState(false);

  const fetchInvoice = () => ({
    invoice: 'lnbc1-sample-invoice-1234567890abcdefg',
    paymentHash: 'abc123',
  });

  const checkPayment = async () => {
    await new Promise(r => setTimeout(r, 2000));
    return { paid: true, pdfUrl: '/stubbed.pdf' };
  };



  const convert = async () => {
    setInvoice(null);
    const { invoice, paymentHash } = fetchInvoice();
    setInvoice(invoice);
    setShowModal(true);
    setStatus('Waiting for payment...');

    setTimeout(async () => {
      const result = await checkPayment(paymentHash);

      if (!result.paid) {
        setStatus('Payment failed. Please try again.');
        return;
      }

      setStatus('Payment received! Downloading PDF...');
      const link = document.createElement('a');
      link.href = result.pdfUrl;
      link.download = 'converted.pdf';
      link.click();

      setStatus('Download Complete!');
    }, 3000);
  };

  const copyInvoice = () => {
    navigator.clipboard.writeText(invoice);
    setStatus('Copied!');
    setTimeout(() => setStatus('Waiting for payment...'), 1000);
  };

  const cancelPayment = () => {
    setShowModal(false);
    setInvoice(null);
    setStatus('');
  };

  return (
    <div>
      <h2 className="text-2xl font-bold mb-2 text-zinc-900 dark:text-zinc-100">Playground</h2>
      <div className="border border-zinc-200 dark:border-zinc-800 rounded-2xl p-6 mt-6">
        <p className="text-sm text-zinc-600 dark:text-zinc-500 mt-0">
        Try it out in our playground! Instantly convert your markdown to a PDF for just $0.01 per file.
        </p>
        <textarea 
          className="w-full h-32 font-sans text-base mb-4 p-3 rounded-2xl border border-zinc-200 dark:border-zinc-700 bg-zinc-50 dark:bg-zinc-900 text-zinc-900 dark:text-zinc-100 placeholder-zinc-400 dark:placeholder-zinc-500 focus:outline-none focus:border-indigo-600 focus:ring-2 focus:ring-indigo-200 dark:focus:border-indigo-400 dark:focus:ring-indigo-900 transition text-sm"
          value={markdown}
          onChange={(e) => setMarkdown(e.target.value)}
          placeholder="Copy and paste your markdown to convert here."
        />
        <button
          onClick={convert}
          disabled={!markdown}
          className="px-6 py-3 rounded-2xl font-semibold transition w-full
            bg-indigo-600 dark:bg-indigo-700 text-white shadow-sm hover:bg-indigo-800 dark:hover:bg-indigo-400
            disabled:bg-zinc-200 dark:disabled:bg-zinc-800 disabled:text-zinc-400 dark:disabled:text-zinc-600
            disabled:opacity-50 disabled:shadow-none disabled:cursor-not-allowed"
        >
          Convert to PDF
        </button>
      </div>

      {showModal && invoice && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/60 dark:bg-black/40 backdrop-blur-sm px-2">
          <div
            className="bg-white dark:bg-zinc-900 p-5 rounded-2xl shadow-2xl w-[95vw] max-w-full sm:max-w-md text-center border border-zinc-200 dark:border-zinc-800 flex flex-col space-y-3 animate-[swipefadeup_0.35s_ease]"
            style={{ minWidth: 0 }}
          >
            <h3 className="text-2xl font-bold text-zinc-900 dark:text-zinc-100 mb-0 mt-1">Scan to Pay</h3>
            <p className="text-sm text-zinc-600 dark:text-zinc-300 mb-0">
              This is your Lightning invoice. Use any wallet that supports Lightning to pay.
            </p>
            <div className="flex justify-center">
              <img
                src={`https://api.qrserver.com/v1/create-qr-code/?size=180x180&data=${encodeURIComponent(invoice)}`}
                alt="Lightning QR code"
                className="rounded-xl border border-zinc-100 dark:border-zinc-800 w-44 h-44"
              />
            </div>
            <div className="flex items-center bg-zinc-50 dark:bg-zinc-800 border border-zinc-200 dark:border-zinc-700 rounded-md px-3 py-2 font-mono text-xs text-zinc-800 dark:text-zinc-200">
              <div className="flex-1 break-all pr-2">{invoice}</div>
              <button
                onClick={copyInvoice}
                className="p-1 rounded hover:bg-indigo-100 dark:hover:bg-indigo-900 transition"
                aria-label="Copy invoice"
              >
                <svg
                  xmlns="http://www.w3.org/2000/svg"
                  width="16"
                  height="16"
                  fill="none"
                  viewBox="0 0 24 24"
                  stroke="currentColor"
                  strokeWidth="2"
                  className="text-indigo-700 dark:text-indigo-600"
                >
                  <rect x="9" y="9" width="13" height="13" rx="2" ry="2" />
                  <path d="M5 15H4a2 2 0 0 1-2-2V4a2 2 0 0 1 2-2h9a2 2 0 0 1 2 2v1" />
                </svg>
              </button>
            </div>
            <div className="text-indigo-700 dark:text-indigo-600 min-h-[1.5em]">{status}</div>
            <button
              onClick={cancelPayment}
              className="w-full px-4 py-2 rounded-2xl bg-zinc-100 dark:bg-zinc-800 text-zinc-700 dark:text-zinc-200 font-medium hover:bg-zinc-200 dark:hover:bg-zinc-700 transition"
            >
              Cancel
            </button>
          </div>
          <style>
            {`
              @keyframes swipefadeup {
                from { opacity: 0; transform: translateY(32px);}
                to { opacity: 1; transform: translateY(0);}
              }
              @keyframes swipefadedown {
                from { opacity: 1; transform: translateY(0);}
                to { opacity: 0; transform: translateY(32px);}
              }
            `}
          </style>
        </div>
      )}
    </div>
  );
};

<MarkdownToPDFPlayground />
